# Overrides for base-game buildings
building_commercial_zone = {
	base_buildtime = @b1_time

	exempt_from_ai_planet_specialization = no
	category = trade

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	planet_modifier = {
		job_clerk_add = 3
	}

	triggered_desc = {
		text = job_clerk_effect_desc
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_policy_flag = lvb_sex_trade_policy_allowed
			}
		}
		modifier = {
			job_lvb_whore_add = 1
			job_clerk_add = -1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_policy_flag = lvb_sex_trade_policy_encouraged
			}
		}
		modifier = {
			job_lvb_whore_add = 2
			job_clerk_add = -2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_active_tradition = tr_mercantile_commercial_enterprise
			}
		}
		modifier = {
			job_merchant_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				has_active_tradition = tr_mercantile_commercial_enterprise
			}
		}
		text = job_merchant_effect_desc
	}

	prerequisites = {
		"tech_interplanetary_commerce"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_policy_flag = lvb_sex_trade_policy_allowed }
		}
		text = building_lvb_commercial_zone_allowed_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_policy_flag = lvb_sex_trade_policy_encouraged }
		}
		text = building_lvb_commercial_zone_encouraged_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				OR = {
					has_policy_flag = lvb_sex_trade_policy_allowed
					has_policy_flag = lvb_sex_trade_policy_encouraged
				}
			}
		}
		text = job_lvb_whore_effect_desc
	}

	upgrades = {
		"building_commercial_megaplex"
	}
}

building_factory_1 = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = manufacturing

	potential = {
		exists = owner
		owner = {
			country_uses_consumer_goods = yes
		}
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
		NOT = { has_building = building_lvb_dairy_farm }
		NOT = { has_building = building_lvb_dairy_facility }
		NOT = { has_building = building_lvb_dairy_arcology }
		OR = {
			owner = {
				is_ai = no
			}
			has_any_industry_district = yes
		}
	}

	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony

			AND = {
				exists = owner
				OR = {
					owner = { country_uses_consumer_goods = no }
					AND = {
						owner = {
							is_ai = yes
						}
						NOT = { has_district = district_rw_industrial }
						NOT = { has_district = district_hab_industrial }
						NOT = { has_district = district_industrial }
					}
				}
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_crafter_empire = no
			}
		}
		modifier = {
			job_artisan_add = @b1_jobs
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_crafter_empire = yes
			}
		}

			exists = owner

			job_artisan_drone_add = @b1_jobs

		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	prerequisites = {
		tech_basic_industry
	}

	upgrades = {
		building_factory_2
	}

	triggered_desc = {
		trigger = {
			exists = owner

				is_regular_empire = yes
				is_crafter_empire = no
			}

	triggered_desc = {
