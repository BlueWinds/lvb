# Overrides for base-game buildings

building_commercial_zone = {
	base_buildtime = 360

	category = trade

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
	}

	allow = {
		buildings_simple_allow = yes
	}

	destroy_trigger = {
		OR = {
		owner = { is_regular_empire = no }
			has_modifier = slave_colony
		}
	}

	planet_modifier = {
		job_clerk_add = 5
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_policy_flag = lvb_sex_trade_policy_allowed
			}
		}
		modifier = {
			job_lvb_whore_add = 1
			job_clerk_add = -1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_policy_flag = lvb_sex_trade_policy_encouraged
			}
		}
		modifier = {
			job_lvb_whore_add = 2
			job_clerk_add = -2
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}
		upkeep = {
			energy = 2
		}
	}

	prerequisites = {
		"tech_interplanetary_commerce"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_policy_flag = lvb_sex_trade_policy_allowed }
		}
		text = building_lvb_commercial_zone_allowed_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_policy_flag = lvb_sex_trade_policy_encouraged }
		}
		text = building_lvb_commercial_zone_encouraged_desc
	}

	triggered_desc = {
		text = job_clerk_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				has_policy_flag = lvb_sex_trade_policy_allowed
				has_policy_flag = lvb_sex_trade_policy_encouraged
			}
		}
		text = job_lvb_whore_effect_desc
	}

	upgrades = {
		"building_commercial_megaplex"
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 5
			planet = {
				OR = {
					has_building = building_holo_theatres
					has_building = building_hyper_entertainment_forum
				}
				OR = {
					num_buildings = { type = building_research_lab_1 value = 3 } #Prevents AI falling behind on research to all but the most research dedicated players
					has_building = building_research_lab_2
					has_building = building_research_lab_3
				}
				NOR = {
					has_building = building_commercial_zone
					has_building = building_commercial_megaplex
				}
			}
		}
	}

	ai_resource_production = {
		energy = 1
		trigger = {
			always = yes
		}
	}
}

building_clinic = {
	base_buildtime = @b1_time
	base_cap_amount = 1

	category = amenity

	potential = {
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			AND = {
				buildings_stop_pop = yes
				owner = { is_ai = yes }
			}
		}
	}

	planet_modifier = {
		job_healthcare_add = 2
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_tradition = tr_lvb_sexualization_bioreactors
			}
		}
		modifier = {
			job_lvb_bioreactor_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	prerequisites = {
		"tech_frontier_health"
	}

	upgrades = {
		"building_hospital"
	}

	triggered_desc = {
		text = job_healthcare_effect_desc
	}

	{{> bioreactor_triggered_desc }}

	ai_weight = {
		weight = 50
		modifier = {
			factor = 0
			OR = {
				is_planet_class = pc_habitat
				buildings_stop_pop = yes
			}
		}
		modifier = {
			factor = 0
			count_pops = { limit = { is_sapient = yes } count > 75 }
		}
	}
}
