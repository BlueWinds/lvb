# Overrides for base-game buildings
building_commercial_zone = {
	base_buildtime = @b1_time

	category = trade

	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
	}

	allow = {
		buildings_simple_allow = yes
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
		}
	}

	planet_modifier = {
		job_clerk_add = 5
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_policy_flag = lvb_sex_trade_policy_allowed
			}
		}
		modifier = {
			job_lvb_whore_add = 1
			job_clerk_add = -1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_policy_flag = lvb_sex_trade_policy_encouraged
			}
		}
		modifier = {
			job_lvb_whore_add = 2
			job_clerk_add = -2
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	prerequisites = {
		"tech_interplanetary_commerce"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_policy_flag = lvb_sex_trade_policy_allowed }
		}
		text = building_lvb_commercial_zone_allowed_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_policy_flag = lvb_sex_trade_policy_encouraged }
		}
		text = building_lvb_commercial_zone_encouraged_desc
	}

	triggered_desc = {
		text = job_clerk_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				OR = {
					has_policy_flag = lvb_sex_trade_policy_allowed
					has_policy_flag = lvb_sex_trade_policy_encouraged
				}
			}
		}
		text = job_lvb_whore_effect_desc
	}

	upgrades = {
		"building_commercial_megaplex"
	}

	ai_resource_production = {
		energy = 2
		trigger = {
			always = yes
		}
	}
	ai_resource_production = {
		energy = 3
		trigger = {
			OR = {
				num_unemployed >= 10
				num_assigned_jobs = {
					job = servant
					value >= 10
				}
			}
		}
	}
}

		text = job_clerk_effect_desc

building_clinic = {
	base_buildtime = @b1_time
	base_cap_amount = 1

	category = amenity

	potential = {
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = slave_colony }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			AND = {
				buildings_stop_pop = yes
				owner = { is_ai = yes }
			}
		}
	}

	planet_modifier = {
		job_healthcare_add = 2
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_tradition = tr_lvb_sexualization_bioreactors
			}
		}
		modifier = {
			job_healthcare_add = -1
			job_lvb_bioreactor_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	prerequisites = {
		"tech_frontier_health"
	}

	upgrades = {
		"building_hospital"
	}

	triggered_desc = {
		text = job_healthcare_effect_desc
	}

	{{> bioreactor_triggered_desc }}
}
