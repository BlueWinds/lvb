	base_buildtime = 180
	category = resource

	potential = {
		exists = owner
		owner = {
			OR = {
				NOT = { has_authority = auth_machine_intelligence }
				has_civic = civic_machine_assimilator
				has_civic = civic_machine_servitor
			}
		}
		NOT = { has_modifier = resort_colony }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				has_authority = auth_machine_intelligence
				NOR = {
					has_civic = civic_machine_assimilator
					has_civic = civic_machine_servitor
				}
			}
			planet = {
				OR = {
					has_modifier = resort_colony
					owner = {
						is_ai = yes
						has_monthly_income = {
							resource = food
							value > 100
						}
					}
				}
			}
		}
	}

	resources = {
        category = planet_buildings
        cost = {
			minerals = 250
		}
        upkeep = {
			energy = 4
		}
	}

	upgrades = {
		building_lvb_dairy_facility
	}

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		modifier = {
			job_lvb_milk_cow_add = 2
		}
	}

	triggered_desc = {
        trigger = {
            always = yes
        }
        text = job_lv_milk_cow_effect_desc
	}

	ai_weight = {
		factor = 0
		modifier = {
			weight = 500
			OR = {
				free_jobs < 3
				lv_unemployed = yes
			}
			owner = {
				lv_uses_food = yes
				has_monthly_income = {
					resource = food
					value < 0
				}
			}
		}
		modifier = {
			weight = 1000
			OR = {
				free_jobs < 3
				lv_unemployed = yes
			}
			OR = {
				owner = {
					lv_uses_food = yes
					has_monthly_income = {
						resource = food
						value < 125
					}
				}
			}
		}
		modifier = {
			factor = 0
			owner = {
				has_monthly_income = {
					resource = food
					value > 1
				}
			}
		}
		modifier = {
			weight = 1
			free_building_slots = 0
		}
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
}
