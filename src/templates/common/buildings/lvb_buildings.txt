building_lvb_dairy_farm = {
	base_buildtime = 180
	category = resource

	potential = {
		exists = owner
		owner = {
			OR = {
				NOT = { has_authority = auth_machine_intelligence }
				has_civic = civic_machine_assimilator
				has_civic = civic_machine_servitor
			}
		}
		NOT = { has_modifier = resort_colony }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				has_authority = auth_machine_intelligence
				NOR = {
					has_civic = civic_machine_assimilator
					has_civic = civic_machine_servitor
				}
			}
			planet = {
				OR = {
					has_modifier = resort_colony
					owner = {
						is_ai = yes
						has_monthly_income = {
							resource = food
							value > 100
						}
					}
				}
			}
		}
	}

	resources = {
        category = planet_buildings
        cost = {
			minerals = 250
		}
        upkeep = {
			energy = 4
		}
	}

	upgrades = {
		building_lvb_dairy_facility
	}

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		modifier = {
			job_lvb_milk_cow_add = 2
		}
	}

	{{> cow_triggered_desc }}

	ai_weight = {
		factor = 0
		modifier = {
			weight = 500
			OR = {
				free_jobs < 3
				lvb_unemployed = yes
			}
			owner = {
				lvb_uses_food = yes
				has_monthly_income = {
					resource = food
					value < 0
				}
			}
		}
		modifier = {
			weight = 1000
			OR = {
				free_jobs < 3
				lvb_unemployed = yes
			}
			OR = {
				owner = {
					lvb_uses_food = yes
					has_monthly_income = {
						resource = food
						value < 125
					}
				}
			}
		}
		modifier = {
			factor = 0
			owner = {
				has_monthly_income = {
					resource = food
					value > 1
				}
			}
		}
		modifier = {
			weight = 1
			free_building_slots = 0
		}
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
}

building_lvb_dairy_facility = {		# Dairy Facility (upgrades from Dairy Farm)
	base_buildtime = 360
	category = resource
	can_build = no

	potential = {
		exists = owner
		owner = {
			OR = {
				NOT = { has_authority = auth_machine_intelligence }
				has_civic = civic_machine_assimilator
				has_civic = civic_machine_servitor
			}
		}
		NOT = { has_modifier = resort_colony }
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				has_authority = auth_machine_intelligence
				NOR = {
					has_civic = civic_machine_assimilator
					has_civic = civic_machine_servitor
				}
			}
			planet = {
				OR = {
					has_modifier = resort_colony
					owner = {
						is_ai = yes
						has_monthly_income = {
							resource = food
							value > 100
						}
					}
				}
			}
		}
	}

	allow = {
		exists = owner
		has_upgraded_capital = yes
	}

	resources = {
        category = planet_buildings
        cost = {
			minerals = 500
			volatile_motes = 50
		}
        upkeep = {
			energy = 5
			volatile_motes = 1
		}
	}

	upgrades = {
		building_lvb_dairy_arcology
	}

	planet_modifier = {
		job_lvb_milk_cow_add = 5
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_tradition = tr_lvb_sexualization_hot_tanks
			}
		}
		modifier = {
			job_lvb_hot_tank_add = 1
		}
	}

	{{> cow_triggered_desc }}

	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			lvb_has_motes = no
		}
		modifier = {
			factor = 0
			owner = { has_country_flag = buildings_new_mote_building }
		}
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
	on_queued = {
		IF = {
			limit = {
				owner = { has_technology = tech_construction_templates }
			}
			owner = { set_timed_country_flag = { flag = buildings_new_mote_building days = 270 } }
		}
		ELSE_IF = {
			limit = {
				owner = { has_technology = tech_assembly_pattern }
			}
			owner = { set_timed_country_flag = { flag = buildings_new_mote_building days = 390 } }
		}
		else = {
			planet = {
				owner = { set_timed_country_flag = { flag = buildings_new_mote_building days = 510 } }
			}
		}
	}
}

building_lvb_slave_brothel = {
	base_buildtime = 360
	base_cap_amount = 1

	category = government

	potential = {
		owner = { is_regular_empire = yes }
		owner = { has_policy_flag = slavery_allowed }
		any_owned_pop = { is_enslaved = yes }
	}

	allow = {
		has_upgraded_capital = yes
	}

	destroy_trigger = { # destroys if true
		OR = {
			owner = { is_regular_empire = no }
			NOT = { owner = { has_policy_flag = slavery_allowed } }
		}
	}

	planet_modifier = {}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 400
		}
		upkeep = { energy = 2 }
	}

	prerequisites = {
		"tech_lvb_sex_trade"
	}

	ai_weight = {
		weight = 5
		modifier = {
			factor = 0
			NOT = { any_owned_pop = { is_domestic_servant = yes } }
		}
	}
}
